    2. `connect_ai/scoring_utils.py`:
       ```python
       """Utilitarios numericos compartilhados entre os pipelines do CONNECT.AI.

       Concentra a conversao distancia coseno -> score 0-100, citada como
       pitfall 11.1 do BRIEFING (ChromaDB retorna distancia, nao similaridade).
       """

       from __future__ import annotations


       def distancia_cosseno_para_score(distancia: float) -> float:
           """Converte distancia coseno (0..2) em score de similaridade (100..0).

           Formula:
             similaridade = 1 - distancia / 2     # mapeia [0, 2] -> [1, 0]
             score        = similaridade * 100    # escala 0-100

           Resultado e sempre truncado em [0.0, 100.0] para tolerar pequenos
           overshoots numericos do ChromaDB (ex: distancia ligeiramente
           negativa por arredondamento).

           Args:
               distancia: Distancia coseno retornada pelo ChromaDB (esperado em [0, 2]).

           Returns:
               Score em [0.0, 100.0] -- 100 quando distancia=0 (vetores identicos),
               50 quando distancia=1 (vetores ortogonais), 0 quando distancia=2
               (vetores opostos).
           """
           similaridade = 1.0 - (float(distancia) / 2.0)
           score = similaridade * 100.0
           if score < 0.0:
               return 0.0
           if score > 100.0:
               return 100.0
           return score
       ```

    ```python
    """Wrapper de repositorio do ChromaDB para o CONNECT.AI.

    Encapsula o cliente nativo, expondo apenas a API necessaria pelos pipelines
    de ingestao (Fase 4) e consumo (Fase 5). Mantem o ChromaDB como detalhe
    de implementacao -- substituir por outro vector DB no futuro afeta apenas
    este arquivo.

    Design:
      - Embeddings sao fornecidos pre-calculados (o repositorio nao chama o Gemini).
      - Persistencia em disco via PersistentClient (diretorio gitignored).
      - Distancia coseno explicita (`hnsw:space=cosine`).
      - Metadados estruturados gravados junto ao vetor para suportar filtros hard.
    """

    from __future__ import annotations

    from dataclasses import dataclass
    from typing import Any, Dict, List, Optional, Sequence

    import chromadb
    from chromadb.config import Settings

    from connect_ai.config import obter_diretorio_chroma, obter_nome_colecao
    from connect_ai.schema import Perfil, construir_documento_semantico


    @dataclass
    class ResultadoBusca:
        """Item retornado por `Repositorio.buscar`.

        Attributes:
            id: ID do perfil armazenado.
            documento: Texto que foi embedado (resultado de construir_documento_semantico).
            metadata: Metadados estruturados gravados (idade, cidade, etc.).
            distancia: Distancia coseno retornada pelo ChromaDB (0 = identico, 2 = oposto).
        """

        id: str
        documento: str
        metadata: Dict[str, Any]
        distancia: float


    class Repositorio:
        """Wrapper do ChromaDB para perfis do CONNECT.AI."""

        def __init__(
            self,
            diretorio: Optional[str] = None,
            nome_colecao: Optional[str] = None,
        ) -> None:
            """Cria/abre a colecao de perfis.

            Args:
                diretorio: Diretorio de persistencia. Default: lido de
                    `CHROMA_PERSIST_DIR` (env) ou "./chroma_db".
                nome_colecao: Nome da colecao. Default: lido de
                    `CHROMA_COLLECTION` (env) ou "perfis_connect_ai".
            """
            self._diretorio = diretorio or obter_diretorio_chroma()
            self._nome_colecao = nome_colecao or obter_nome_colecao()
            self._cliente = chromadb.PersistentClient(
                path=self._diretorio,
                settings=Settings(anonymized_telemetry=False, allow_reset=True),
            )
            self._colecao = self._cliente.get_or_create_collection(
                name=self._nome_colecao,
                metadata={"hnsw:space": "cosine"},
            )

        # ------------------------------------------------------------------ #
        # Construcao de metadados
        # ------------------------------------------------------------------ #

        @staticmethod
        def _metadata_de_perfil(perfil: Perfil) -> Dict[str, Any]:
            """Extrai metadados estruturados (REPO-03) do Perfil.

            Apenas tipos primitivos (str, int, float, bool) -- requirement do
            ChromaDB para campos de metadata.
            """
            faixa_min, faixa_max = perfil.faixa_etaria_pref
            return {
                "idade": int(perfil.idade),
                "cidade": str(perfil.cidade),
                "genero": str(perfil.genero),
                "genero_preferido": str(perfil.genero_preferido),
                "faixa_etaria_min": int(faixa_min),
                "faixa_etaria_max": int(faixa_max),
                "objetivo": str(perfil.objetivo),
                "nome": str(perfil.nome),  # util para exibicao no front
            }

        # ------------------------------------------------------------------ #
        # API publica
        # ------------------------------------------------------------------ #

        def inserir(
            self,
            perfil: Perfil,
            embedding: Sequence[float],
        ) -> None:
            """Insere (ou atualiza por upsert) um perfil ja embedado.

            Idempotente: re-inserir o mesmo `perfil.id` substitui o registro
            existente, sem duplicar (suporte a ING-04 da Fase 4).

            Args:
                perfil: Instancia validada de Perfil.
                embedding: Vetor pre-calculado pelo text-embedding-004 (768d esperado).
            """
            documento = construir_documento_semantico(perfil)
            metadata = self._metadata_de_perfil(perfil)
            self._colecao.upsert(
                ids=[perfil.id],
                embeddings=[list(embedding)],
                metadatas=[metadata],
                documents=[documento],
            )

        def inserir_lote(
            self,
            perfis: Sequence[Perfil],
            embeddings: Sequence[Sequence[float]],
        ) -> int:
            """Insere multiplos perfis em uma chamada.

            Args:
                perfis: Sequencia de Perfil.
                embeddings: Sequencia de vetores na mesma ordem.

            Returns:
                Quantidade de perfis inseridos.

            Raises:
                ValueError: Se as duas sequencias tiverem tamanhos diferentes.
            """
            if len(perfis) != len(embeddings):
                raise ValueError(
                    f"Numero de perfis ({len(perfis)}) difere do numero de "
                    f"embeddings ({len(embeddings)})."
                )
            if not perfis:
                return 0
            ids = [p.id for p in perfis]
            documentos = [construir_documento_semantico(p) for p in perfis]
            metadatas = [self._metadata_de_perfil(p) for p in perfis]
            self._colecao.upsert(
                ids=ids,
                embeddings=[list(e) for e in embeddings],
                metadatas=metadatas,
                documents=documentos,
            )
            return len(perfis)

        def buscar(
            self,
            embedding_query: Sequence[float],
            n_resultados: int = 30,
            filtros: Optional[Dict[str, Any]] = None,
        ) -> List[ResultadoBusca]:
            """Busca vetorial Top-K com filtros hard opcionais via metadados.

            Args:
                embedding_query: Vetor da query.
                n_resultados: Quantos vizinhos retornar (Fase 5 usa K=30).
                filtros: Dict no formato `where` do ChromaDB (ex:
                    `{"objetivo": "namoro", "cidade": {"$eq": "Sao Paulo"}}`).
                    None = sem filtros.

            Returns:
                Lista de ResultadoBusca ordenada por distancia crescente
                (mais similar primeiro).
            """
            kwargs: Dict[str, Any] = {
                "query_embeddings": [list(embedding_query)],
                "n_results": n_resultados,
            }
            if filtros:
                kwargs["where"] = filtros
            resposta = self._colecao.query(**kwargs)

            ids = (resposta.get("ids") or [[]])[0]
            documentos = (resposta.get("documents") or [[]])[0]
            metadatas = (resposta.get("metadatas") or [[]])[0]
            distancias = (resposta.get("distances") or [[]])[0]

            return [
                ResultadoBusca(
                    id=ids[i],
                    documento=documentos[i] if i < len(documentos) else "",
                    metadata=dict(metadatas[i]) if i < len(metadatas) and metadatas[i] else {},
                    distancia=float(distancias[i]),
                )
                for i in range(len(ids))
            ]

        def contar(self) -> int:
            """Retorna o numero de documentos na colecao."""
            return int(self._colecao.count())

        def existe(self, id_perfil: str) -> bool:
            """Verifica se um perfil esta presente na colecao por id."""
            resultado = self._colecao.get(ids=[id_perfil])
            ids = resultado.get("ids") or []
            return id_perfil in ids

        def resetar(self) -> None:
            """Apaga todos os documentos da colecao (recria do zero).

            Util para testes e para o botao "Repopular banco" do front (Fase 6).
            """
            try:
                self._cliente.delete_collection(name=self._nome_colecao)
            except Exception:
                # Se a colecao nao existir, segue para recria-la
                pass
            self._colecao = self._cliente.get_or_create_collection(
                name=self._nome_colecao,
                metadata={"hnsw:space": "cosine"},
            )
    ```
